
can_private_construction = {
	OR = {
		AND = {
			owner = { is_player = yes }
			has_game_rule = infres_private_construction_allowed
		}
		AND = {
			has_game_rule = infres_player_and_ai_allowed
			has_game_rule = infres_private_construction_allowed
		}
	} 
}
can_gov_construction = {
	OR = {
		owner = { is_player = yes }
		has_game_rule = infres_player_and_ai_allowed
	}
}
