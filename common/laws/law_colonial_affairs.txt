law_no_colonial_affairs = {
	group = lawgroup_colonization
	icon = "gfx/interface/icons/law_icons/no_colonial_affairs.dds"
	progressiveness = 0	
	pop_support = { value = 0 }
	ai_will_do = { has_law = law_type:law_council_republic }	
	ai_enact_weight_modifier = {	# Petitions
		value = 0
		if = { limit = { has_journal_entry = je_government_petition has_variable = desired_law_var scope:law = var:desired_law_var } add = 750 }
		if = {
			limit = { has_law = law_type:law_council_republic }
			add = 100
		}
		if = {
			limit = { has_strategy = ai_strategy_anti_imperialism }
			multiply = 2
		}
	}
}
law_frontier_colonization = {
	group = lawgroup_colonization
	icon = "gfx/interface/icons/law_icons/frontier_colonization.dds"
	progressiveness = 25
	unlocking_technologies = { colonization }
	can_enact = {
		OR = {
			any_scope_state = {
				is_homeland_of_country_cultures = ROOT
				any_neighbouring_state = {
					OR = {
						owner = { is_country_type = decentralized }
						AND = {
							owner = { THIS = ROOT }
							NOT = { is_homeland_of_country_cultures = ROOT }
						}
					}
				}
			}
			is_country_type = colonial
		}
	}
	limited_to_frontier_colonization = yes
	institution = institution_colonial_affairs
	modifier = { state_migration_pull_unincorporated_mult	= 1 }	# Changed - It's much easier to convince someone to move to a new state if they can do it by themselves.
	# institution_modifier = { state_migration_pull_unincorporated_mult	= 0.2 }
	pop_support = {
		value = 0
		add = {	# Farmers like frontiers
			desc = "POP_FARMERS"			
			if = {
				limit = { is_pop_type = farmers }
				value = 0.25
			}
		}
		add = {	# Officers like colonialism
			desc = "POP_OFFICERS"			
			if = {
				limit = {
					owner = { has_law = law_type:law_no_colonial_affairs } 
					is_pop_type = officers
				}
				value = 0.25
			}
	}
	ai_will_do = {
		NOR = {
			has_law = law_type:law_colonial_exploitation
			has_law = law_type:law_colonial_resettlement
		}
	}
}
