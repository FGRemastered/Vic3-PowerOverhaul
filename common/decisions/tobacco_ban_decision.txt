# Ban
ban_tobacco = {
	is_shown = {
		exists = market
		any_primary_culture = { has_discrimination_trait = german_speaking }
		NOT = { is_banning_goods = g:tobacco }
	}
	possible = {
		any_primary_culture = { has_discrimination_trait = german_speaking }
		exists = market
		market.owner = ROOT
	}
	when_taken = { effect_tobacco_ban = yes }
    ai_chance = {
        base = 0
		modifier = {
			add = 10
			trigger = { authority >= 200 }
		}
		modifier = {
			add = 10
			trigger = {
				any_interest_group = {
					limit = {
						OR = {
							is_being_bolstered = yes
							is_in_government = yes
						}
						OR = {
							leader = { has_ideology = ideology_fascist }
							is_interest_group_type = ig_petty_bourgeoisie
						}
					}
				}
			}
		}
    }
}
# Renew
ban_tobacco_again = {
	is_shown = {
		is_banning_goods = g:tobacco
		exists = market
	}
	possible = {
		market.owner = ROOT
		is_banning_goods = g:tobacco
		market = {
			OR = {
				any_scope_country = { NOT = { is_banning_goods = g:tobacco } }
				any_scope_country = { any_primary_culture = { has_cultural_obsession = tobacco } }
				any_scope_country = { any_scope_state = { has_building = building_tobacco_plantation } }
				any_scope_country = { any_scope_state = { has_building = building_infres_tobacco_plantation } }
			}
		}
	}
	when_taken = { effect_tobacco_ban = yes }
    ai_chance = {
        base = 0

			add = 10

		}
		modifier = {
			add = 10

				any_interest_group = {
					limit = {
						OR = {
							is_being_bolstered = yes
							is_in_government = yes
						}

							leader = { has_ideology = ideology_fascist }

		}
    }
}
# Rescind
remove_tobacco_ban = {
	is_shown = {
		is_banning_goods = g:tobacco
		exists = market
	}
	possible = { market.owner = ROOT }
	when_taken = { effect_remove_tobacco_ban = yes }
	ai_chance = {
		base = 0
		modifier = {
			trigger = { authority < 50 }
			add = 10
		}
		modifier = {
			trigger = {
				any_interest_group = {
					limit = {
						OR = {
							is_being_bolstered = yes
							is_in_government = yes
						}
						OR = {
							leader = { has_ideology = ideology_fascist }
							is_interest_group_type = ig_petty_bourgeoisie
						}
					}
				}
			}
			add = -20
		}
   }
}
